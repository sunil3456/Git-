Create a git folder
Commands: 

touch foldername
git init
git config --global user.name "name"
git config --global user.email "email id"
git config --list
ls -a (to check hidden filles)
git status
git add filename1 filename2 filename3
git commit -m "first commit" 
git log
git reset head filename
git reset --soft parentcommitID
git show commitID
git add ./*/-A (to add all)
git clone centralhuburl
git log -2/ --oneline/ --oneline -2
git log --author=name
git log --since=2016-09-07
git log --until=yyyy-mm-dd
